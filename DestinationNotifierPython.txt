def send_mail(latitude, longitude, to = "1by20ai001@bmsit.in", emg=0):

def isfloat(num):
    try:
        float(num)
        return True
    except ValueError:
        return False


def calc_distance(lat2, lon2):
    dlon = lon2 - lon1
    dlat = lat2 - lat1
    a = sin(dlat / 2)**2 + cos(lat1) * cos(lat2) * sin(dlon / 2)**2
    c = 2 * asin(sqrt(a))
    r = 6371

    return c * r
